Função

 Descrição

 Exemplo

 ABS

 Retorna o valor absoluto de um número.

 SELECT ABS(-10);

 ACOS

 Retorna o arco cosseno de um número.

 SELECT ACOS(0.5);

 ADDDATE

 Adiciona um intervalo de tempo a uma data.

 SELECT ADDDATE('2024-05-01', INTERVAL 7 DAY);

 ADDTIME

 Adiciona um intervalo de tempo a uma hora.

 SELECT ADDTIME('12:00:00', '01:30:00');

 ASCII

 Retorna o código ASCII do primeiro caractere de uma string.

 SELECT ASCII('A');

 ASIN

 Retorna o arco seno de um número.

 SELECT ASIN(0.5);

 ATAN

 Retorna a tangente inversa de um número.

 SELECT ATAN(1);

 AVG

 Retorna a média dos valores em uma coluna.

 SELECT AVG(coluna_numerica) FROM tabela;

 BIT_AND

 Retorna a operação AND bit a bit de todos os valores na coluna.

 SELECT BIT_AND(coluna_numerica) FROM tabela;

 BIT_OR

 Retorna a operação OR bit a bit de todos os valores na coluna.

 SELECT BIT_OR(coluna_numerica) FROM tabela;

 BIT_XOR

 Retorna a operação XOR bit a bit de todos os valores na coluna.

 SELECT BIT_XOR(coluna_numerica) FROM tabela;

 CEIL ou CEILING

 Arredonda um número para o próximo número inteiro maior ou igual.

 SELECT CEIL(10.3);

 CHAR_LENGTH

 Retorna o número de caracteres em uma string.

 SELECT CHAR_LENGTH('Hello');

 CHARACTER_LENGTH

 Retorna o número de caracteres em uma string.

 SELECT CHARACTER_LENGTH('Hello');

 COALESCE

 Retorna o primeiro valor não nulo em uma lista de expressões.

 SELECT COALESCE(coluna1, coluna2, coluna3) FROM tabela;

 CONCAT

 Concatena duas ou mais strings.

 SELECT CONCAT(coluna1, ' ', coluna2) FROM tabela;

CONCAT_WS

 Concatena duas ou mais strings com um separador.

 SELECT CONCAT_WS(',', coluna1, coluna2, coluna3) FROM tabela;

CONVERT

 Converte uma string de um conjunto de caracteres para outro.

 SELECT CONVERT('texto', CHARSET);

COS

 Retorna o cosseno de um número.

 SELECT COS(1);

COT

 Retorna a tangente cotangente de um número.

 SELECT COT(1);

COUNT

 Retorna o número de linhas em um conjunto de resultados.

 SELECT COUNT(*) FROM tabela;

CURDATE

 Retorna a data atual.

 SELECT CURDATE();

CURRENT_DATE

 Retorna a data atual.

 SELECT CURRENT_DATE();

CURRENT_TIME

 Retorna a hora atual.

 SELECT CURRENT_TIME();

CURRENT_TIMESTAMP

 Retorna a data e a hora atuais.

 SELECT CURRENT_TIMESTAMP();

CURTIME

Retorna a hora atual.

SELECT CURTIME();

DATE

Extrai a parte da data de uma data ou data/hora.

SELECT DATE(data_hora) FROM tabela;

 DATE_ADD

 Adiciona um intervalo de tempo a uma data.

 SELECT DATE_ADD(data, INTERVAL 1 MONTH) FROM tabela;

 DATE_FORMAT

 Formata uma data conforme o especificado.

 SELECT DATE_FORMAT(data, '%d/%m/%Y') FROM tabela;

 DATE_SUB

 Subtrai um intervalo de tempo de uma data.

 SELECT DATE_SUB(data, INTERVAL 1 MONTH) FROM tabela;

 DATEDIFF

 Retorna o número de dias entre duas datas.

 SELECT DATEDIFF(data_fim, data_inicio) AS diferenca_dias FROM tabela;

 DAY

 Extrai o dia de uma data.

 SELECT DAY(data) FROM tabela;

 DAYNAME

 Retorna o nome do dia da semana de uma data.

 SELECT DAYNAME(data) FROM tabela;

 DAYOFMONTH

 Retorna o dia do mês de uma data.

 SELECT DAYOFMONTH(data) FROM tabela;

 DAYOFWEEK

 Retorna o dia da semana de uma data.

 SELECT DAYOFWEEK(data) FROM tabela;

 DAYOFYEAR

 Retorna o dia do ano de uma data.

 SELECT DAYOFYEAR(data) FROM tabela;

 DEGREES

 Converte radianos em graus.

 SELECT DEGREES(PI()) FROM tabela;

 EXP

 Retorna e elevado à potência de .

 SELECT EXP(2);

 EXTRACT

 Extrai uma parte de uma data.

 SELECT EXTRACT(YEAR FROM data) FROM tabela;

 FIELD

 Retorna a posição de uma string em uma lista de strings.

 SELECT FIELD('valor', 'valor1', 'valor2', 'valor3') FROM tabela;

 FIND_IN_SET

 Retorna a posição de uma string em uma lista de strings.

 SELECT FIND_IN_SET('valor', 'valor1,valor2,valor3') FROM tabela;

 FLOOR

 Arredonda um número para o próximo número inteiro menor ou igual.

 SELECT FLOOR(10.7);

 FORMAT

 Formata um número.

 SELECT FORMAT(coluna_numerica, 2) FROM tabela;

 FROM_DAYS

 Converte um número de dias em uma data.

 SELECT FROM_DAYS(737059);

 HOUR

 Extrai a hora de uma data ou data/hora.

 SELECT HOUR(data_hora) FROM tabela;

 IF

 Retorna um valor se uma condição for verdadeira e outro se for falsa.

 SELECT IF(coluna1 > coluna2, 'Maior', 'Menor') FROM tabela;

 IFNULL

 Retorna um valor se a expressão for nula, caso contrário, retorna a expressão.

 SELECT IFNULL(coluna, 'Substituto') FROM tabela;

 INSERT

 Insere uma substring em uma string em uma posição especificada.

 SELECT INSERT('abcdef', 3, 2, '123') FROM tabela;

 INSTR

 Retorna a posição da primeira ocorrência de uma substring em uma string.

 SELECT INSTR('exemplo', 'mp') FROM tabela;

 ISNULL

 Verifica se uma expressão é nula.

 SELECT ISNULL(coluna) FROM tabela;

 LAST_DAY

 Retorna o último dia do mês para uma data.

 SELECT LAST_DAY(data) FROM tabela;

 LCASE ou LOWER

 Converte uma string em minúsculas.

 SELECT LOWER('TEXTO') FROM tabela;

 LEAST

 Retorna o menor valor da lista de expressões.

 SELECT LEAST(coluna1, coluna2, coluna3) FROM tabela;

 LEFT

 Retorna os caracteres mais à esquerda de uma string.

 SELECT LEFT('abcde', 3) FROM tabela;

 LENGTH

 Retorna o número de caracteres em uma string.

 SELECT LENGTH('Hello');

 LINESTRING

 Cria um objeto LineString a partir de coordenadas.

 SELECT LINESTRING(1, 1, 2, 2, 3, 3) FROM tabela;

 LN

 Retorna o logaritmo natural de um número.

 SELECT LN(2);

 LOCALTIME

 Retorna a hora atual.

 SELECT LOCALTIME();

 LOCALTIMESTAMP

 Retorna a data e a hora atuais.

 SELECT LOCALTIMESTAMP();

 LOG

 Retorna o logaritmo de um número em uma base específica.

 SELECT LOG(10, 100);

 LOG10

 Retorna o logaritmo de um número na base 10.

 SELECT LOG10(100);

 LOG2

 Retorna o logaritmo de um número na base 2.

 SELECT LOG2(8);

 LPAD

 Preenche uma string à esquerda com outra string até um comprimento especificado.

 SELECT LPAD('hello', 10, '*') FROM tabela;

 LTRIM

 Remove os espaços em branco do início de uma string.

 SELECT LTRIM('   example   ');

 MAKE_SET

 Retorna um conjunto de bits que representa uma lista de strings.

 SELECT MAKE_SET(1, 'valor1', 'valor2', 'valor3') FROM tabela;

 MID ou SUBSTRING_INDEX

 Retorna uma parte de uma string a partir de uma posição e com uma contagem de caracteres especificados.

 SELECT MID('abcdef', 2, 3) FROM tabela;

 MINUTE

 Extrai os minutos de uma hora.

 SELECT MINUTE('12:34:56');

 MOD

 Retorna o resto da divisão de um número por outro.

 SELECT MOD(10, 3);

 MONTH

 Extrai o mês de uma data.

 SELECT MONTH('2024-05-01');

 MONTHNAME

 Retorna o nome do mês de uma data.

 SELECT MONTHNAME('2024-05-01');

 NOW

 Retorna a data e a hora atuais.

 SELECT NOW();

 NULLIF

 Retorna nulo se duas expressões forem iguais, caso contrário, retorna a primeira expressão.

 SELECT NULLIF(coluna1, coluna2) FROM tabela;

 OCT

 Retorna a representação octal de um número.

 SELECT OCT(10);

 OCTET_LENGTH

 Retorna o número de bytes em uma string.

 SELECT OCTET_LENGTH('Hello');

 ORD

 Retorna o código Unicode do primeiro caractere de uma string.

 SELECT ORD('A');

 PI

 Retorna o valor de pi (p).

 SELECT PI();

 POSITION

 Retorna a posição de uma substring em uma string.

 SELECT POSITION('lo' IN 'Hello');

 POW ou POWER

 Retorna um número elevado à potência de outro número.

 SELECT POWER(2, 3);

 QUARTER

 Retorna o trimestre de uma data.

 SELECT QUARTER('2024-05-01');

 RADIANS

 Converte graus em radianos.

 SELECT RADIANS(180);

 RAND

 Retorna um número decimal aleatório entre 0 e 1.

 SELECT RAND();

 RANDOM_BYTES

 Retorna uma string de bytes aleatórios.

 SELECT RANDOM_BYTES(10);

 RELEASE_LOCK

 Libera um bloqueio nomeado.

 SELECT RELEASE_LOCK('nome_bloqueio');

 REPLACE

 Substitui todas as ocorrências de uma substring em uma string por outra substring.

 SELECT REPLACE('abab', 'a', 'b');

 REVERSE

 Inverte uma string.

 SELECT REVERSE('abcdef');

 RIGHT

 Retorna os caracteres mais à direita de uma string.

 SELECT RIGHT('abcdef', 3);

 ROUND

 Arredonda um número para um número especificado de casas decimais.

 SELECT ROUND(10.456, 2);

 RPAD

 Preenche uma string à direita com outra string até um comprimento especificado.

 SELECT RPAD('hello', 10, '*') FROM tabela;

 SEC_TO_TIME

 Converte segundos em um valor de tempo.

 SELECT SEC_TO_TIME(3661);

 SECOND

 Extrai os segundos de uma hora.

 SELECT SECOND('12:34:56');

 SESSION_USER

 Retorna o nome de usuário e o host para a conexão atual.

 SELECT SESSION_USER();

 SIGN

 Retorna o sinal de um número.

 SELECT SIGN(10);

 SIN

 Retorna o seno de um número.

 SELECT SIN(1);

 SPACE

 Retorna uma string de espaços.

 SELECT SPACE(5);

 SQRT

 Retorna a raiz quadrada de um número.

 SELECT SQRT(16);

 STR_TO_DATE

 Converte uma string em uma data.

 SELECT STR_TO_DATE('01/05/2024', '%d/%m/%Y');

 SUBDATE

 Subtrai um intervalo de tempo de uma data.

 SELECT SUBDATE(data, INTERVAL 1 MONTH) FROM tabela;

 SUBSTRING ou SUBSTR

 Retorna uma parte de uma string.

 SELECT SUBSTRING('abcdef', 2, 3);

 SUBSTRING_INDEX ou MID

 Retorna uma parte de uma string a partir de uma posição e com uma contagem de caracteres especificados.

 SELECT SUBSTRING_INDEX('www.mysql.com', '.', 2);

 SUBTIME

 Subtrai um intervalo de tempo de uma hora.

 SELECT SUBTIME('12:00:00', '01:30:00');

 SUM

 Retorna a soma dos valores em uma coluna.

 SELECT SUM(coluna_numerica) FROM tabela;

 SYSDATE

 Retorna a data e a hora atuais.

 SELECT SYSDATE();

 TAN

 Retorna a tangente de um número.

 SELECT TAN(1);

 TIME_TO_SEC

 Converte um valor de tempo em segundos.

 SELECT TIME_TO_SEC('12:34:56');

TIMESTAMPDIFF

 Retorna a diferença entre duas datas ou tempos.

SELECT TIMESTAMPDIFF(SECOND, '2022-01-01 00:00:00', '2022-01-02 12:00:00');

 TO_DAYS

 Converte uma data em dias desde a data 0.

 SELECT TO_DAYS('2024-05-01');

 TO_SECONDS

 Converte uma data e hora em segundos desde a data 0.

 SELECT TO_SECONDS('2024-05-01 12:34:56');

 TRIM

 Remove espaços em branco do início e do final de uma string.

 SELECT TRIM('   exemplo   ');

 UCASE ou UPPER

 Converte uma string em maiúsculas.

 SELECT UPPER('texto');

 UNIX_TIMESTAMP

 Retorna a data e hora atual como um timestamp UNIX.

 SELECT UNIX_TIMESTAMP();

 WEEK

 Retorna o número da semana de uma data.

 SELECT WEEK('2024-05-01');

 WEEKDAY

 Retorna o dia da semana de uma data.

 SELECT WEEKDAY('2024-05-01');

 YEAR

 Retorna o ano de uma data.

 SELECT YEAR('2024-05-01');
