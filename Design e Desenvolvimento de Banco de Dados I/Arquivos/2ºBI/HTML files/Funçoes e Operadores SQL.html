Comparativos

=

Igualdade

Verificar se dois valores são iguais

SELECT * FROM tabela WHERE coluna = valor;

<> ou !=

Desigualdade

Verificar se dois valores são diferentes

SELECT * FROM tabela WHERE coluna <> valor;

> 

Maior que

Verificar se um valor é maior que outro

SELECT * FROM tabela WHERE coluna > valor;

< 

Menor que

Verificar se um valor é menor que outro

SELECT * FROM tabela WHERE coluna < valor;

>=

Maior ou igual

Verificar se um valor é maior ou igual a outro

SELECT * FROM tabela WHERE coluna >= valor;

<=

Menor ou igual

Verificar se um valor é menor ou igual a outro

SELECT * FROM tabela WHERE coluna <= valor;

BETWEEN

Intervalo inclusivo

Verificar se um valor está dentro de um intervalo específico

SELECT * FROM tabela WHERE coluna BETWEEN valor1 AND valor2;

IN

Correspondência com uma lista de valores

Verificar se um valor corresponde a qualquer valor em uma lista

SELECT * FROM tabela WHERE coluna IN (valor1, valor2, ...);

LIKE

Correspondência de padrões

Verificar se um valor corresponde a um padrão específico

SELECT * FROM tabela WHERE coluna LIKE 'padrão';

Lógicas

AND

Retorna verdadeiro se todas as condições forem verdadeiras

Verificar se todas as luzes da casa estão ligadas

SELECT (1 > 0) AND (2 > 1);

OR

Retorna verdadeiro se pelo menos uma condição for verdadeira

Verificar se pelo menos um sinal de trânsito está verde

SELECT (sinal1 = 'verde') OR (sinal2 = 'verde');

NOT

Inverte o valor de verdadeiro para falso e vice-versa

Determinar se o céu não está nublado

SELECT NOT (clima = 'nublado');

XOR

Retorna verdadeiro se exatamente uma condição for verdadeira e todas as outras forem falsas

Verificar se somente uma lâmpada está acesa em um quarto

SELECT XOR(lampada1, lampada2);

IS NULL

Verifica se uma expressão é NULL

Verificar se uma caixa está vazia

SELECT nome FROM caixa WHERE conteudo IS NULL;

IS NOT NULL

Verifica se uma expressão não é NULL

Verificar se uma caixa não está vazia

SELECT nome FROM caixa WHERE conteudo IS NOT NULL;

IFNULL

Se a primeira expressão for NULL, retorna a segunda expressão; caso contrário, retorna a primeira expressão

Usado para substituir um valor NULL por um valor alternativo

SELECT IFNULL(coluna1, 'Valor alternativo');

CASE

Avalia uma lista de condições e retorna um resultado correspondente à primeira condição verdadeira, ou uma condição ELSE se nenhuma condição for verdadeira

Como um grande switch case em programação

SELECT CASE WHEN pontuacao > 90 THEN 'A'
WHEN pontuacao > 80 THEN 'B'
ELSE 'C'
END AS 'Nota' FROM alunos;

UNIÃO

UNION

Combina o resultado de duas ou mais consultas em uma única tabela

Combinar duas listas de alunos em uma única lista

SELECT coluna FROM tabela1 UNION SELECT coluna FROM tabela2;

UNION ALL

Combina o resultado de duas ou mais consultas em uma única tabela, incluindo duplicatas

Combinar duas listas de alunos, mantendo todas as entradas

SELECT coluna FROM tabela1 UNION ALL SELECT coluna FROM tabela2;

INTERSECT

Retorna as linhas que ocorrem em ambos os resultados das consultas

Identificar alunos presentes em ambas as listas

SELECT coluna FROM tabela1 INTERSECT SELECT coluna FROM tabela2;

EXCEPT

Retorna as linhas que ocorrem no primeiro resultado, mas não no segundo

Identificar alunos presentes apenas na primeira lista

SELECT coluna FROM tabela1 EXCEPT SELECT coluna FROM tabela2;
